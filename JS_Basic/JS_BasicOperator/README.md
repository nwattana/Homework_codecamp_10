# แบบฝึกหัด การดำเนินการเบื้องต้น
## 1. ให้ระบุค่าของ a, b, c , d และ e หลังจากจบ statements ทั้งหมด
``` javascript
{
    let a = 1;
    let b = 2;
    let c = a++;
    let d = ++c;
    let e = ++d + d++ + d;
}
```
> a = 2 , b = 2 , c = 1, d = 3, e = 4 + 4 + 5
